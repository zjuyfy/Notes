              \subsection{NFS网络共享文件系统}

              \hl{安装}
              sudo apt-get install nfs-kernel-server\\
              \hl{配置NFS}
              sudo vi /etc/exports\\
              \hl{设置NFS目录}
              在打开的文件末尾添加<目录路径> *(rw,sync,no\_root\_squash)\\
              \hl{重启NFS}sudo service nfs-kernel-server restart\\
              \hl{在从机挂载文件系统}mount -t nfs -o nolock <主机ip>:<主机目录> <从机目录>
